{
	atomic&lt;int&gt; shared_count;
	atomic&lt;int&gt; weak_count;
};

template&lt;typename T&gt;
class SharedPtr
{
	T* object;
	Counter *  counter;
};

template&lt;typename T&gt;
class WeakPtr
{
	T* object;
	Counter* counter;
};

void test()
{
	SharedPtr&lt;int&gt; s1,s2;
	WeakPtr&lt;int&gt; w1;

	s1 = s2;// 如果是两个SP，则导致 shared_count++
	w1 = s1;// 导致weak_count++ ，当然这里还会导致原来的计数器weak_count--，当shared_count和weak_count同时为0的时候，释放计数器
}
